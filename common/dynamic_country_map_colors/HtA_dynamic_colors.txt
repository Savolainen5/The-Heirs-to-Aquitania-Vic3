high_kingdom_ireland = {
	color = "gaelic_green"
	
	possible = {
		exists = c:IRE
		THIS = c:IRE
		OR = {
			owns_entire_state_region = STATE_LOWLANDS
			owns_entire_state_region = STATE_HIGHLANDS
		}
		NOT = { exists = c:SCO }
	}
}

basic_ireland = {
	color = "irish_green"
	#color = "naples_purple" (for testing purposes)
	
	possible = {
		exists = c:IRE
		THIS = c:IRE
		NOT = {
			owns_entire_state_region = STATE_HIGHLANDS
			owns_entire_state_region = STATE_LOWLANDS
		}
	}
}
