je_objective_expand_goods_production = {
	should_be_pinned_by_default = yes
	icon = "gfx/interface/icons/objectives/economic_dominance.dds"

	complete = {
		scope:journal_entry = { is_goal_complete = yes }
		scope:target = {
			cash_reserves_ratio > 0.25
			weekly_profit > 0
			occupancy > 0.75
		}
	}		
	
	current_value = {
		value = scope:target.level
	}
	
	goal_add_value = {
		value = 1
	}
	
	is_progressing = {
		exists = scope:target
		scope:target = { is_under_construction = yes }
	}
	
	invalid = {
		custom_tooltip = {
			text = je_objective_expand_goods_production_invalid_tooltip
			OR = {
				NOT = { exists = scope:target }
				NOT = { scope:target.owner = root }
			}
		}
	}
	
	
	on_complete = {
		hidden_effect = {
			owner = {
				start_tutorial_lesson = { 
					tutorial_lesson = expand_goods_production_complete
					journal_entry = je:je_objective_expand_goods_production
				}
			}
		}
	}

	on_invalid = {
		remove_active_objective_subgoal = sg_expand_goods_production
	}
		
	weight = 1000
	inheritable = yes
}

je_specialized_inputs = {
	should_be_pinned_by_default = yes
	icon = "gfx/interface/icons/objectives/economic_dominance.dds"

	immediate = {	
	}
	
	complete = {
		OR = {
			any_scope_building = {
				filter = {
					OR = {
						is_building_type = building_rye_farm
						is_building_type = building_wheat_farm
						is_building_type = building_maize_farm
						is_building_type = building_millet_farm
					}
				}
				percent >= 0.75
				OR = {
					has_active_production_method = pm_tractors_building_rye_farm
					has_active_production_method = pm_tractors_building_wheat_farm
					has_active_production_method = pm_tractors_building_maize_farm
					has_active_production_method = pm_tractors_building_millet_farm
					has_active_production_method = pm_compression_ignition_tractors_building_rye_farm
					has_active_production_method = pm_compression_ignition_tractors_building_wheat_farm
					has_active_production_method = pm_compression_ignition_tractors_building_maize_farm
					has_active_production_method = pm_compression_ignition_tractors_building_millet_farm
				}
				weekly_profit > 0
				occupancy >= 0.5
			}
			any_scope_building = {
				filter = {
					OR = {
						is_building_type = building_coal_mine
						is_building_type = building_iron_mine
						is_building_type = building_lead_mine
						is_building_type = building_sulfur_mine
					}
				}
				percent >= 0.75
				OR = {
					has_active_production_method = pm_steam_donkey_building_coal_mine
					has_active_production_method = pm_steam_donkey_mine
				}
				weekly_profit > 0
				occupancy >= 0.5
			}
			any_scope_building = {
				is_building_type = building_railway
				or = {
					level >= 5
				}
			}
		}
	}

	on_complete = {
		set_variable = player_obj_home_focus
	}
	
	invalid = {
		has_variable = player_obj_export_focus
	}

	weight = 1002
	inheritable = yes
}
