	save_scope_as = second_country
	if = {
		limit = {
			has_strategic_adjacency = scope:first_country
		}
		
		random_list = {			
			35 = {
				set_secret_goal = {
					country = scope:first_country
					secret_goal = befriend
				}			
			}
			65 = {
				set_secret_goal = {
					country = scope:first_country
					secret_goal = none
				}			
			}	
		}
	}	
	else = {
		set_secret_goal = {
			country = scope:first_country
			secret_goal = none
		}			
	}	
	
	scope:first_country = {
		if = {
			limit = {
				has_strategic_adjacency = scope:second_country
			}

				35 = {

					set_secret_goal = {
						country = scope:second_country
						secret_goal = none
					}			
				}	
			}
		}

		}		
	}
}
